# 指南

# 1.安装组件库

npm i xinnong-ui

# 2.引用组件库

//全部引入
import 'xinnong-ui/dist/css/index.css'
import MUI from 'xinnong-ui'
Vue.use(MUI)

//按需引入
import 'xinnong-ui/dist/css/demo.css'
import Demo from 'xinnong-ui'
Vue.use(Demo)
